const catwayService = require("../services/catwayService");

/**
 * Contrôleur pour récupérer tous les catways
 * @param {Object} req - Requête Express
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie la liste des catways en JSON (status 200)
 */
const getAllCatways = async (req, res) => {
    try {
        const catways = await catwayService.getAllCatways();
        res.status(200).json(catways);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

/**
 * Contrôleur pour récupérer un catway par son ID
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.id - ID du catway
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie le catway en JSON (status 200) ou erreur (status 404)
 */
const getCatwayById = async (req, res) => {
    try {

        const id = req.params.id;
        const catway = await catwayService.getCatwayById(id);
        res.status(200).json(catway);

    } catch (error) {
        res.status(404).json({ error: error.message });
    }
};

/**
 * Contrôleur pour créer un nouveau catway
 * @param {Object} req - Requête Express
 * @param {Object} req.body - Corps de la requête
 * @param {Object} req.body.catwayNumber - Numéro du catway
 * @param {Object} req.body.type - Type du catway
 * @param {Object} req.body.catwayState - État du catway
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie le catway créé en JSON (status 201) ou erreur (status 500)
 */
const createCatway = async (req, res) => {
    try {
        const catwayData = req.body;
        const newCatway = await catwayService.createCatway(catwayData);
        res.status(201).json(newCatway);

    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

/**
 * Contrôleur pour mettre à jour complètement un catway
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.id - ID du catway à modifier
 * @param {Object} req.body - Nouvelles données du catway
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie le catway mis à jour en JSON (status 200) ou erreur (status 500)
 */
const updateCatway = async (req, res) => {
    try {

        const id = req.params.id;
        const catwayData = req.body;
        const updatedCatway = await catwayService.updateCatway(id, catwayData);
        res.status(200).json(updatedCatway);

    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

/**
 * Contrôleur pour modifier partiellement un catway
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.id - ID du catway à modifier
 * @param {Object} req.body - Champs à modifier (tous optionnels)
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie le catway modifié en JSON (status 200) ou erreur (status 500)
 */
const patchCatway = async (req, res) => {
    try {
        const id = req.params.id;
        const catwayData = req.body;
        const patchedCatway = await catwayService.patchCatway(id, catwayData);
        res.status(200).json(patchedCatway);

    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

/**
 * Contrôleur pour supprimer un catway
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.id - ID du catway à supprimer
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie un message de confirmation en JSON (status 200) ou erreur (status 500)
 */
const deleteCatway = async (req, res) => {
    try {
        const id = req.params.id;
        const deletedCatway = await catwayService.deleteCatway(id);
        res.status(200).json(deletedCatway);

    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};
