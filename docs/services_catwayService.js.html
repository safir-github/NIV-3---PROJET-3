const Catway = require("../models/catway");

/**
 * Récupère tous les catways de la base de données
 * @returns {Promise<Array<Object>>} Liste de tous les catways
 * @throws {Error} Erreur lors de la récupération des catways
 */
const getAllCatways = async () => {
    try {
        const catways = await Catway.find();
        return catways;
    } catch (error) {
        throw error;
    }
};

/**
 * Récupère un catway par son identifiant
 * @param {string} id - Identifiant MongoDB du catway
 * @returns {Promise<Object>} Le catway trouvé
 * @throws {Error} "Catway non trouvé" si l'ID n'existe pas
 * @throws {Error} Erreur lors de la recherche
 */
const getCatwayById = async (id) => {
    try {
        const catway = await Catway.findById(id);

        if (!catway) {
            throw new Error("Catway non trouvé");
        }

        return catway;
    } catch (error) {
        throw error;
    }
};

/**
 * Crée un nouveau catway dans la base de données
 * @param {Object} catwayData - Données du catway à créer
 * @param {number} catwayData.catwayNumber - Numéro unique du catway
 * @param {string} catwayData.type - Type du catway ('long' ou 'short')
 * @param {string} catwayData.catwayState - État du catway
 * @returns {Promise<Object>} Le catway créé avec son _id
 * @throws {Error} Erreur lors de la création (validation, doublon, etc.)
 */
const createCatway = async (catwayData) => {
    try {
        const newCatway = new Catway({
            catwayNumber: catwayData.catwayNumber,
            type: catwayData.type,
            catwayState: catwayData.catwayState
        });

        await newCatway.save();
        return newCatway;
    } catch (error) {
        throw error;
    }
};

/**
 * Met à jour complètement un catway (remplace tous les champs)
 * @param {string} id - Identifiant MongoDB du catway à modifier
 * @param {Object} catwayData - Nouvelles données du catway
 * @param {number} catwayData.catwayNumber - Nouveau numéro du catway
 * @param {string} catwayData.type - Nouveau type du catway ('long' ou 'short')
 * @param {string} catwayData.catwayState - Nouvel état du catway
 * @returns {Promise<Object>} Le catway mis à jour
 * @throws {Error} "Catway non trouvé" si l'ID n'existe pas
 * @throws {Error} Erreur lors de la mise à jour
 */
const updateCatway = async (id, catwayData) => {
    try {
        const catway = await Catway.findById(id);

        if (!catway) {
            throw new Error("Catway non trouvé");
        }

        const updateData = {
            catwayNumber: catwayData.catwayNumber,
            type: catwayData.type,
            catwayState: catwayData.catwayState
        };

        const updatedCatway = await Catway.findByIdAndUpdate(id, updateData, { new: true });
        return updatedCatway;
    } catch (error) {
        throw error;
    }
};

/**
 * Modifie partiellement un catway (seulement les champs fournis)
 * @param {string} id - Identifiant MongoDB du catway à modifier
 * @param {Object} catwayData - Champs à modifier (tous optionnels)
 * @param {number} [catwayData.catwayNumber] - Nouveau numéro du catway (optionnel)
 * @param {string} [catwayData.type] - Nouveau type du catway (optionnel)
 * @param {string} [catwayData.catwayState] - Nouvel état du catway (optionnel)
 * @returns {Promise<Object>} Le catway mis à jour
 * @throws {Error} "Catway non trouvé" si l'ID n'existe pas
 * @throws {Error} Erreur lors de la modification
 */
const patchCatway = async (id, catwayData) => {
    try {
        const catway = await Catway.findById(id);

        if (!catway) {
            throw new Error("Catway non trouvé");
        }

        const updateData = {};

        if (catwayData.catwayNumber !== undefined) {
            updateData.catwayNumber = catwayData.catwayNumber;
        }

        if (catwayData.type !== undefined) {
            updateData.type = catwayData.type;
        }

        if (catwayData.catwayState !== undefined) {
            updateData.catwayState = catwayData.catwayState;
        }

        const updatedCatway = await Catway.findByIdAndUpdate(id, updateData, { new: true });
        return updatedCatway;
    } catch (error) {
        throw error;
    }
};

/**
 * Supprime un catway de la base de données
 * @param {string} id - Identifiant MongoDB du catway à supprimer
 * @returns {Promise<Object>} Message de confirmation
 * @returns {string} return.message - "Catway supprimé"
 * @throws {Error} "Catway non trouvé" si l'ID n'existe pas
 * @throws {Error} Erreur lors de la suppression
 */
const deleteCatway = async (id) => {
    try {
        const catway = await Catway.findById(id);

        if (!catway) {
            throw new Error("Catway non trouvé");
        }

        await Catway.findByIdAndDelete(id);
        return { message: "Catway supprimé" };
    } catch (error) {
        throw error;
    }
};

    patchCatway,
    deleteCatway,
};
