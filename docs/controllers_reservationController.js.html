const reservationService = require("../services/reservationService");

/**
 * Contrôleur pour récupérer toutes les réservations d'un catway
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.catwayNumber - Numéro du catway
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie la liste des réservations en JSON (status 200) ou erreur (status 500)
 */
const getReservationsByCatway = async (req, res) => {
    try {
        const catwayNumber = parseInt(req.params.catwayNumber);
        const reservations = await reservationService.getReservationsByCatway(catwayNumber);
        res.status(200).json(reservations);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

/**
 * Contrôleur pour récupérer une réservation par son ID
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.id - ID de la réservation
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie la réservation en JSON (status 200) ou erreur (status 404)
 */
const getReservationById = async (req, res) => {
    try {
        const id = req.params.id;
        const reservation = await reservationService.getReservationById(id);
        res.status(200).json(reservation);
    } catch (error) {
        res.status(404).json({ error: error.message });
    }
};

/**
 * Contrôleur pour créer une nouvelle réservation pour un catway
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.catwayNumber - Numéro du catway
 * @param {Object} req.body - Corps de la requête
 * @param {string} req.body.clientName - Nom du client
 * @param {string} req.body.boatName - Nom du bateau
 * @param {string} req.body.checkIn - Date d'arrivée
 * @param {string} req.body.checkOut - Date de départ
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie la réservation créée en JSON (status 201) ou erreur (status 500)
 */
const createReservation = async (req, res) => {
    try {
        const catwayNumber = parseInt(req.params.catwayNumber);
        const reservationData = {
            ...req.body,
            catwayNumber
        };
        const newReservation = await reservationService.createReservation(reservationData);
        res.status(201).json(newReservation);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

/**
 * Contrôleur pour supprimer une réservation
 * @param {Object} req - Requête Express
 * @param {Object} req.params - Paramètres de route
 * @param {string} req.params.id - ID de la réservation
 * @param {Object} res - Réponse Express
 * @returns {Promise<void>} Envoie un message de confirmation en JSON (status 200) ou erreur (status 404)
 */
const deleteReservation = async (req, res) => {
    try {
        const id = req.params.id;
        const result = await reservationService.deleteReservation(id);
        res.status(200).json(result);
    } catch (error) {
        res.status(404).json({ error: error.message });
    }
};

module.exports = {
    getReservationsByCatway,
    getReservationById,
    createReservation,
    deleteReservation,
};
